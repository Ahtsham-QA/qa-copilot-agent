// LoginPage.js - Page Object Model for the SauceDemo Login Page

class LoginPage {
  constructor(page) {
    this.page = page;

    // Selectors using data-test attributes (preferred) and fallbacks
    this.usernameInput = page.locator('[data-test="username"]');
    this.passwordInput = page.locator('[data-test="password"]');
    this.loginButton = page.locator('[data-test="login-button"]');
    this.errorMessage = page.locator('[data-test="error"]');
    this.errorMessageContainer = page.locator('.error-message-container');
  }

  // Navigate to the SauceDemo login page
  async navigate() {
    await this.page.goto('https://www.saucedemo.com');
  }

  // Enter username/email into the username field
  async enterUsername(username) {
    await this.usernameInput.clear();
    await this.usernameInput.fill(username);
  }

  // Enter password into the password field
  async enterPassword(password) {
    await this.passwordInput.clear();
    await this.passwordInput.fill(password);
  }

  // Click the login button to submit the form
  async clickLoginButton() {
    await this.loginButton.click();
  }

  // Full login action combining username, password, and submit
  async login(username, password) {
    await this.enterUsername(username);
    await this.enterPassword(password);
    await this.clickLoginButton();
  }

  // Get the text content of the error message
  async getErrorMessageText() {
    return await this.errorMessage.textContent();
  }

  // Get the type attribute of the password input field
  async getPasswordFieldType() {
    return await this.passwordInput.getAttribute('type');
  }
}

module.exports = LoginPage;
